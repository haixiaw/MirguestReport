\newsavebox{\createalgheader}
\begin{lrbox}{\createalgheader}
\begin{lstlisting}
#ifndef HelloWorldAlg_hh
#define HelloWorldAlg_hh

#include "SniperKernel/AlgBase.h"

class HelloWorldAlg: public AlgBase {
    public:
        HelloWorldAlg(const std::string& name);
        ~HelloWorldAlg();

        bool initialize();
        bool execute();
        bool finalize();
};

#endif
\end{lstlisting}
\end{lrbox}
